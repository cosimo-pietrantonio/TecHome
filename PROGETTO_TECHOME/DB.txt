CREATE TABLE accessi (

  IDA INT(4) PRIMARY KEY AUTO_INCREMENT,
  Acc_Soci INT(4) REFERENCES soci(Id_Socio),
  Acc_Clienti INT(4) REFERENCES clienti(Id_Cliente),
  ClientePrec INT(4) NOT NULL,
  SociPrec INT(4) NOT NULL, 
  ON UPDATE CASCADE
  ON DELETE CASCADE

) 

CREATE TABLE clienti (
  Id_Cliente INT(4) PRIMARY KEY AUTO_INCREMENT,
  Nome_Cliente varchar(30) NOT NULL,
  Cognome_Cliente varchar(30) NOT NULL,
  Email_Cliente varchar(30) NOT NULL,
  Password_Cliente varchar(12) NOT NULL
)


CREATE TABLE soci (
  Id_Socio INT(4) PRIMARY KEY AUTO_INCREMENT,
  Email_Socio varchar(30) NOT NULL,
  Nome_Socio varchar(30) NOT NULL,
  Cognome_Socio varchar(30) NOT NULL,
  P_Iva_Socio varchar(50) NOT NULL,
  Password_Soci varchar(12) NOT NULL
)
